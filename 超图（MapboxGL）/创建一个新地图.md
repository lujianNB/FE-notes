# 使用引入的依赖创建一个新地图

```
import mapboxgl from 'mapbox-gl'
import '@supermap/iclient-mapboxgl'

let map = new mapboxgl.Map({
  container: 'map',
  style: {
    "version": 8,
    "sources": {
      "raster-tiles": {
        "type": "raster",
        "tiles": ['https://iserver.supermap.io/iserver/services/map-china400/rest/maps/China/zxyTileImage.png?z={z}&x={x}&y={y}'],
        "tileSize": 256
      }
    },
    "layers": [{
      "id": "simple-tiles",
      "type": "raster",
      "source": "raster-tiles",
      "minzoom": 0,
      "maxzoom": 22
    }]
  },
  center: [120.12, 30.16], // 地图初始中心点
  zoom: 10 // 地图初始缩放级别
})
```
